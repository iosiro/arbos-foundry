70/// Helper function to deploy stylus contract from artifact code.
71/// Uses CREATE2 scheme if salt specified.
72fn deploy_stylus_code(
73    ccx: &mut CheatsCtxt,
74    executor: &mut dyn CheatcodesExecutor,
75    path: &str,
76    constructor_args: Option<&Bytes>,
77    value: Option<U256>,
78    salt: Option<U256>,
79) -> Result {
80    let bytecode = get_artifact_code(ccx.state, path, false)?.to_vec();
81
82    let scheme =
83        if let Some(salt) = salt { CreateScheme::Create2 { salt } } else { CreateScheme::Create };
84
85    let create_value = if constructor_args.is_some() {
86        // if constructor args are provided, we need to deploy the contract with value
87        U256::ZERO
88    } else {
89        // if no constructor args, we can deploy without value
90        value.unwrap_or(U256::ZERO)
91    };
92
93    let outcome = executor.exec_create(
94        CreateInputs {
95            caller: ccx.caller,
96            scheme,
97            value: create_value,
98            init_code: bytecode.into(),
99            gas_limit: ccx.gas_limit,
100        },
101        ccx,
102    )?;
103
104    if !outcome.result.result.is_ok() {
105        return Err(crate::Error::from(outcome.result.output));
106    }
107
108    let address = outcome.address.ok_or_else(|| fmt_err!("contract creation failed"))?;
109
110    if let Some(constructor_args) = constructor_args {
111        // cast sig 'stylus_constructor()' => 0x5585258d
112        let mut calldata = vec![0x55, 0x85, 0x25, 0x8d];
113        calldata.extend_from_slice(constructor_args);
114
115        let outcome = executor.exec_call(
116            CallInputs {
117                input: revm::interpreter::CallInput::Bytes(calldata.into()),
118                return_memory_offset: 0..0,
119                gas_limit: outcome.gas().remaining(),
120                bytecode_address: address,
121                target_address: address,
122                caller: ccx.caller,
123                value: revm::interpreter::CallValue::Transfer(value.unwrap_or(U256::ZERO)),
124                scheme: CallScheme::Call,
125                is_static: false,
126            },
127            ccx,
128        )?;
129
130        if !outcome.result.result.is_ok() {
131            return Err(crate::Error::from(outcome.result.output));
132        }
133    }
134
135    Ok(address.abi_encode())
136}

138/// Returns the path to the json artifact depending on the input
139///
140/// Can parse following input formats:
141/// - `path/to/artifact.wasm`
142/// - `path/to/artifact.wasm.br`
143fn get_artifact_code(state: &Cheatcodes, path: &str, compress: bool) -> Result<Bytes> {
144    let path = if path.ends_with(".wasm") || path.ends_with(".wasm.br") {
145        PathBuf::from(path)
146    } else {
147        bail!("invalid artifact path")
148    };
149
150    let path = state.config.ensure_path_allowed(path, FsAccessKind::Read)?;
151    let artifact = fs::read(path)?;
152
153    // if it starts with WASM magic number, we assume it's a Stylus artifact and we first
154    // need to compress it and then prefix it with Stylus discriminant
155    let artifact = if artifact.starts_with(&[0x00, 0x61, 0x73, 0x6d]) && compress {
156        // Compress the artifact if it is a Stylus artifact
157        if let Ok(compressed) =
158            stylus::brotli::compress(&artifact, 11, 22, stylus::brotli::Dictionary::StylusProgram)
159        {
160            compressed
161        } else {
162            bail!("failed to compress stylus artifact")
163        }
164    } else {
165        artifact
166    };
167
168    // If the artifact is already prefixed with Stylus discriminant, we can return it as is.
169    // Otherwise, we need to prefix it with Stylus discriminant and compression level.
170
171    let artifact = if artifact.starts_with(arbos_revm::constants::STYLUS_DISCRIMINANT) {
172        // If the artifact is already prefixed with Stylus discriminant, we can return it as is.
173        artifact
174    } else {
175        // If not, we need to prefix it with Stylus discriminant and compression level.
176        [arbos_revm::constants::STYLUS_DISCRIMINANT, &[compress as u8], artifact.as_ref()].concat()
177    };
178
179    // add init code
180    let artifact = get_init_code_of_empty_constructor(artifact);
181
182    Ok(Bytes::from(artifact))
183}

185fn get_init_code_of_empty_constructor(bytecode: Vec<u8>) -> Vec<u8> {
186    let mut init = Vec::with_capacity(2 + bytecode.len() + 4);
187
188    // step 1: fixed header
189    init.extend_from_slice(hex!("608060405234801561001057600080fd5b50").as_slice());
190
191    // step 2: push the bytecode length (always 2 bytes)
192    init.extend_from_slice(&[0x61]);
193
194    let bytecode_len = bytecode.len();
195
196    // always returns at least 32 bytes
197    let bytecode_len_bytes = bytecode_len.to_be_bytes();
198
199    // can safely ignore the first 30 bytes as length is always less than 2^16
200    init.extend_from_slice(&bytecode_len_bytes[bytecode_len_bytes.len() - 2..]);
201
202    // step 3: fixed footer
203    init.extend_from_slice(hex!("806100206000396000f3fe").as_slice());
204
205    // push the bytecode itself
206    init.extend_from_slice(&bytecode);
207
208    init
209}
