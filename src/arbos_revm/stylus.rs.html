22pub fn build_evm_data<CTX>(context: &mut CTX, input: InputsImpl) -> EvmData
23where
24    CTX: ArbitrumContextTr,
25{
26    // find target_address in context.evm.journaled_state.call_stack excluding last
27    // if found, set reentrant to true
28    // else set reentrant to false
29    // let reentrant = if context
30    //     .evm
31    //     .journaled_state
32    //     .call_stack
33    //     .iter()
34    //     .filter(|&x| *x == self.inputs.target_address)
35    //     .count()
36    //     > 1
37    // {
38    //     1
39    // } else {
40    //     0
41    // };
42    let reentrant = 0;
43
44    let config_env = context.cfg();
45    let arbos_env = context.chain();
46
47    let block_env = context.block();
48    let tx_env = context.tx();
49
50    let base_fee = block_env.basefee();
51
52    let evm_data: EvmData = EvmData {
53        arbos_version: arbos_env.arbos_version() as u64,
54        block_basefee: Bytes32::from(U256::from(base_fee).to_be_bytes()),
55        chainid: config_env.chain_id(),
56        block_coinbase: Bytes20::try_from(block_env.beneficiary().as_slice()).unwrap(),
57        block_gas_limit: U64::wrapping_from(block_env.gas_limit()).to::<u64>(),
58        block_number: U64::wrapping_from(block_env.number()).to::<u64>(),
59        block_timestamp: U64::wrapping_from(block_env.timestamp()).to::<u64>(),
60        contract_address: Bytes20::try_from(input.target_address.as_slice()).unwrap(),
61        module_hash: Bytes32::try_from(keccak256(input.target_address.as_slice()).as_slice())
62            .unwrap(),
63        msg_sender: Bytes20::try_from(input.caller_address.as_slice()).unwrap(),
64        msg_value: Bytes32::try_from(input.call_value.to_be_bytes_vec()).unwrap(),
65        tx_gas_price: Bytes32::from(
66            U256::from(tx_env.effective_gas_price(base_fee as u128)).to_be_bytes(),
67        ),
68        tx_origin: Bytes20::try_from(tx_env.caller().as_slice()).unwrap(),
69        reentrant,
70        return_data_len: 0,
71        cached: true,
72        tracing: true,
73    };
74
75    evm_data
76}
