194    /// Common method to build API requestor for both inspected and non-inspected modes
195    fn build_api_requestor(
196        &mut self,
197        input: InputsImpl,
198        is_static: bool,
199        request_handler: impl Fn(
200            &mut Self,
201            InputsImpl,
202            bool,
203            EvmApiMethod,
204            Vec<u8>,
205        ) -> (Vec<u8>, VecReader, ArbGas),
206    ) -> EvmApiRequestor<VecReader, StylusHandler> {
207        let evm = Arc::new(Mutex::new(self));
208
209        let callback = {
210            let evm = evm.clone();
211            let inputs = input.clone();
212
213            move |req_type: arbutil::evm::api::EvmApiMethod,
214                  req_data: Vec<u8>|
215                  -> (Vec<u8>, VecReader, arbutil::evm::api::Gas) {
216                let mut evm = evm.lock().unwrap();
217                request_handler(&mut evm, inputs.clone(), is_static, req_type, req_data)
218            }
219        };
220
221        let callback: EvmApiHandler<'_> = Arc::new(Box::new(callback));
222        let unsafe_callback: &'static EvmApiHandler<'_> = unsafe { mem::transmute(&callback) };
223        EvmApiRequestor::new(StylusHandler::new(unsafe_callback.clone()))
224    }

226    /// Extract common Stylus execution context from frame input
227    fn extract_stylus_context(&mut self) -> Option<(StylusExecutionContext, B256, Bytes)> {
228        let frame_input = {
229            let frame = self.frame_stack().get();
230            match frame.input {
231                FrameInput::Call(ref input) => input.clone(),
232                _ => return None,
233            }
234        };
235
236        let bytecode_address = frame_input.bytecode_address;
237
238        let (code_hash, bytecode) = {
239            let ctx = self.ctx();
240            if let Ok(code_hash) = ctx.journal_mut().code_hash(bytecode_address) {
241                let bytecode = ctx.journal_mut().code(bytecode_address).unwrap().data;
242                (code_hash.data, bytecode)
243            } else {
244                return None;
245            }
246        };
247
248        let calldata = match &frame_input.input {
249            CallInput::Bytes(calldata) => calldata.clone(),
250            CallInput::SharedBuffer(range) => {
251                if let Some(slice) = self.ctx().local().shared_memory_buffer_slice(range.clone()) {
252                    Bytes::from(slice.to_vec())
253                } else {
254                    Bytes::new()
255                }
256            }
257        };
258
259        let context = StylusExecutionContext {
260            target_address: frame_input.target_address,
261            caller_address: frame_input.caller,
262            call_value: frame_input.value.get(),
263            is_static: frame_input.is_static,
264            gas_limit: frame_input.gas_limit,
265            calldata,
266        };
267
268        Some((context, code_hash, bytecode))
269    }

271    /// Compile Stylus bytecode
272    fn compile_stylus_bytecode(
273        bytecode: &Bytes,
274        code_hash: B256,
275        stylus_env: &dyn ArbitrumChainInfoTr,
276        _gas_limit: u64,
277        compile_config: &CompileConfig,
278    ) -> Result<(Vec<u8>, Module, StylusData), ()> {
279        if let Some(bytecode) = bytecode.strip_prefix(STYLUS_DISCRIMINANT) {
280            let (dictionary, compressed_bytecode) =
281                if let Some((dictionary, compressed_bytecode)) = bytecode.split_at_checked(1) {
282                    (dictionary, compressed_bytecode)
283                } else {
284                    return Err(());
285                };
286
287            let dictionary = match dictionary[0] {
288                0x00 => Dictionary::Empty,
289                0x01 => Dictionary::StylusProgram,
290                _ => unreachable!(),
291            };
292
293            let bytecode = brotli::decompress(compressed_bytecode, dictionary)
294                .or_else(|err| {
295                    if dictionary == Dictionary::Empty {
296                        Ok(compressed_bytecode.to_vec())
297                    } else {
298                        Err(err)
299                    }
300                })
301                .unwrap();
302
303            let mut activation_gas = 10_000_000u64;
304            let (module, stylus_data) = native::activate(
305                bytecode.as_slice(),
306                &Bytes32::from(code_hash.0),
307                stylus_env.stylus_version(),
308                stylus_env.arbos_version() as u64,
309                128,
310                false,
311                &mut activation_gas,
312            )
313            .unwrap();
314
315            let bytecode = native::compile(
316                bytecode.as_slice(),
317                compile_config.version,
318                false,
319                wasmer_types::compilation::target::Target::default(),
320            )
321            .unwrap();
322
323            return Ok((bytecode, module, stylus_data));
324        }
325
326        Err(())
327    }

329    /// Core Stylus execution logic shared between inspected and non-inspected modes
330    fn execute_stylus_program(
331        &mut self,
332        stylus_ctx: StylusExecutionContext,
333        code_hash: B256,
334        bytecode: Bytes,
335        api_request_handler: impl Fn(
336            &mut Self,
337            InputsImpl,
338            bool,
339            EvmApiMethod,
340            Vec<u8>,
341        ) -> (Vec<u8>, VecReader, ArbGas),
342    ) -> Option<InterpreterAction> {
343        let context = self.ctx();
344
345        let stylus_env = context.chain();
346
347        let compile_config =
348            CompileConfig::version(stylus_env.stylus_version(), stylus_env.debug_mode());
349        let stylus_config = StylusConfig::new(
350            stylus_env.stylus_version(),
351            stylus_env.max_depth(),
352            stylus_env.ink_price(),
353        );
354
355        let (serialized, _module, stylus_data) = {
356            let mut cache = PROGRAM_CACHE.lock().unwrap();
357            if let Ok((serialized, module, stylus_data)) =
358                cache.try_get_or_insert(code_hash, || {
359                    Self::compile_stylus_bytecode(
360                        &bytecode,
361                        code_hash,
362                        stylus_env,
363                        stylus_ctx.gas_limit,
364                        &compile_config,
365                    )
366                })
367            {
368                (serialized.clone(), module.clone(), *stylus_data)
369            } else {
370                return None;
371            }
372        };
373
374        let inputs = InputsImpl {
375            target_address: stylus_ctx.target_address,
376            caller_address: stylus_ctx.caller_address,
377            input: revm::interpreter::CallInput::Bytes(Bytes::from(stylus_ctx.calldata.to_vec())),
378            call_value: stylus_ctx.call_value,
379            bytecode_address: Some(stylus_ctx.target_address),
380        };
381
382        let evm_data = build_evm_data(self.ctx(), inputs.clone());
383        let evm_api =
384            self.build_api_requestor(inputs.clone(), stylus_ctx.is_static, api_request_handler);
385
386        let mut instance = unsafe {
387            NativeInstance::deserialize(serialized.as_slice(), compile_config, evm_api, evm_data)
388                .unwrap()
389        };
390
391        let mut gas = Gas::new(stylus_ctx.gas_limit);
392        if !gas.record_cost(stylus_data.init_cost as u64) {
393            return Some(InterpreterAction::Return(InterpreterResult {
394                result: InstructionResult::OutOfGas,
395                output: Default::default(),
396                gas: Default::default(),
397            }));
398        }
399
400        let ink_limit = stylus_config.pricing.gas_to_ink(arbutil::evm::api::Gas(gas.remaining()));
401
402        gas.spend_all();
403
404        let bytecode = match inputs.input() {
405            CallInput::Bytes(bytes) => bytes,
406            CallInput::SharedBuffer(_) => todo!(),
407        };
408
409        let outcome = match instance.run_main(bytecode, stylus_config, ink_limit) {
410            Err(e) | Ok(UserOutcome::Failure(e)) => UserOutcome::Failure(e.wrap_err("call failed")),
411            Ok(outcome) => outcome,
412        };
413
414        let mut gas_left = stylus_config.pricing.ink_to_gas(instance.ink_left().into()).0;
415
416        let (kind, data) = outcome.into_data();
417
418        let result = match kind {
419            UserOutcomeKind::Success => revm::interpreter::InstructionResult::Return,
420            UserOutcomeKind::Revert => revm::interpreter::InstructionResult::Revert,
421            UserOutcomeKind::Failure => revm::interpreter::InstructionResult::Revert,
422            UserOutcomeKind::OutOfInk => revm::interpreter::InstructionResult::OutOfGas,
423            UserOutcomeKind::OutOfStack => {
424                gas_left = 0;
425                revm::interpreter::InstructionResult::StackOverflow
426            }
427        };
428
429        gas.erase_cost(gas_left);
430
431        Some(InterpreterAction::Return(InterpreterResult { result, output: data.into(), gas }))
432    }

444    /// Handle contract calls (ContractCall, DelegateCall, StaticCall)
445    fn handle_contract_call(
446        &mut self,
447        input: InputsImpl,
448        is_static: bool,
449        req_type: EvmApiMethod,
450        data: Vec<u8>,
451        call_handler: impl FnOnce(
452            &mut Self,
453            FrameInit,
454        ) -> Result<
455            FrameResult,
456            ContextError<<<CTX as ContextTr>::Db as Database>::Error>,
457        >,
458    ) -> (Vec<u8>, VecReader, ArbGas) {
459        let mut data = data;
460        let bytecode_address = buffer::take_address(&mut data);
461        let value = buffer::take_u256(&mut data);
462        let gas_left = buffer::take_u64(&mut data);
463        let gas_limit = buffer::take_u64(&mut data);
464        let calldata = buffer::take_rest(&mut data);
465
466        let is_static = matches!(req_type, EvmApiMethod::StaticCall) || is_static;
467        let (target_address, caller) = if matches!(req_type, EvmApiMethod::DelegateCall) {
468            (input.target_address, input.caller_address)
469        } else {
470            (bytecode_address, input.target_address)
471        };
472
473        if is_static && !value.is_zero() {
474            return (Status::WriteProtection.into(), VecReader::new(vec![]), ArbGas(gas_left));
475        }
476
477        let gas_limit = if self
478            .ctx()
479            .cfg()
480            .spec()
481            .into()
482            //.into_eth_spec()
483            .is_enabled_in(SpecId::TANGERINE)
484        {
485            min(gas_left - gas_left / 64, gas_limit)
486        } else {
487            gas_limit
488        };
489
490        let mut gas = Gas::new(gas_limit);
491        gas.spend_all();
492
493        let first_frame_input = FrameInput::Call(Box::new(CallInputs {
494            input: CallInput::Bytes(calldata),
495            return_memory_offset: 0..0,
496            gas_limit,
497            bytecode_address,
498            target_address,
499            caller,
500            value: revm::interpreter::CallValue::Transfer(value),
501            scheme: revm::interpreter::CallScheme::Call,
502            is_static,
503        }));
504
505        let next_action = InterpreterAction::NewFrame(first_frame_input);
506
507        let frame_result: Result<_, ContextError<<<CTX as ContextTr>::Db as Database>::Error>> =
508            self.0.frame_stack.get().process_next_action(&mut self.0.ctx, next_action);
509
510        let original_frame_stack = mem::replace(&mut self.0.frame_stack, FrameStack::new());
511
512        if let Ok(ItemOrResult::Item(frame_init)) = frame_result {
513            let result = call_handler(self, frame_init);
514
515            self.0.frame_stack = original_frame_stack;
516            self.0.frame_stack().get().interpreter.memory.free_child_context();
517
518            if let Ok(FrameResult::Call(call_outcome)) = result {
519                gas.erase_cost(call_outcome.gas().remaining());
520                return (
521                    Status::Success.into(),
522                    VecReader::new(call_outcome.output().to_vec()),
523                    ArbGas(gas.spent()),
524                );
525            }
526        }
527
528        (Status::Failure.into(), VecReader::new(vec![]), ArbGas(gas.spent()))
529    }

531    /// Handle contract creation (Create1, Create2)
532    fn handle_contract_creation(
533        &mut self,
534        input: InputsImpl,
535        is_static: bool,
536        req_type: EvmApiMethod,
537        data: Vec<u8>,
538        call_handler: impl FnOnce(
539            &mut Self,
540            FrameInit,
541        ) -> Result<
542            FrameResult,
543            ContextError<<<CTX as ContextTr>::Db as Database>::Error>,
544        >,
545    ) -> (Vec<u8>, VecReader, ArbGas) {
546        let is_create_2 = matches!(req_type, EvmApiMethod::Create2);
547        let mut data = data;
548        let gas_remaining = buffer::take_u64(&mut data);
549        let value = buffer::take_u256(&mut data);
550        let salt = is_create_2.then(|| buffer::take_u256(&mut data));
551        let init_code = buffer::take_rest(&mut data);
552
553        let spec = self.ctx().cfg().spec().into();
554
555        if is_static {
556            return (
557                [vec![0x00], "write protection".as_bytes().to_vec()].concat(),
558                VecReader::new(vec![]),
559                ArbGas(0),
560            );
561        }
562
563        let error_response = (
564            [vec![0x01], Address::ZERO.to_vec()].concat(),
565            VecReader::new(vec![]),
566            ArbGas(gas_remaining),
567        );
568
569        if is_create_2 && !spec.is_enabled_in(SpecId::PETERSBURG) {
570            return error_response;
571        }
572
573        let mut gas_cost = 0;
574        let len = init_code.len();
575
576        if len != 0 && spec.is_enabled_in(SpecId::SHANGHAI) {
577            let max_initcode_size = self.ctx().cfg().max_code_size().saturating_mul(2);
578            if len > max_initcode_size {
579                return error_response;
580            }
581            gas_cost = initcode_cost(len);
582        }
583
584        let scheme = if is_create_2 {
585            if let Some(check_cost) = create2_cost(len).and_then(|cost| gas_cost.checked_add(cost))
586            {
587                gas_cost = check_cost;
588            } else {
589                return error_response;
590            };
591            CreateScheme::Create2 { salt: salt.unwrap() }
592        } else {
593            gas_cost += CREATE;
594            CreateScheme::Create
595        };
596
597        if gas_remaining < gas_cost {
598            return (
599                [vec![0x00], "out of gas".as_bytes().to_vec()].concat(),
600                VecReader::new(vec![]),
601                ArbGas(gas_remaining),
602            );
603        }
604
605        let gas_limit = gas_remaining - gas_cost;
606
607        let gas_stipend = if spec.is_enabled_in(SpecId::TANGERINE) { gas_limit / 64 } else { 0 };
608
609        let mut gas = Gas::new(gas_limit);
610        _ = gas.record_cost(gas_stipend);
611
612        let first_frame_input = FrameInput::Create(Box::new(CreateInputs {
613            caller: input.target_address,
614            scheme,
615            value,
616            init_code,
617            gas_limit: gas.remaining(),
618        }));
619
620        gas.spend_all();
621
622        let next_action = InterpreterAction::NewFrame(first_frame_input);
623
624        let frame_result: Result<_, ContextError<<<CTX as ContextTr>::Db as Database>::Error>> =
625            self.0.frame_stack.get().process_next_action(&mut self.0.ctx, next_action);
626
627        let original_frame_stack = mem::replace(&mut self.0.frame_stack, FrameStack::new());
628
629        if let Ok(ItemOrResult::Item(frame_init)) = frame_result {
630            let result = call_handler(self, frame_init);
631
632            self.0.frame_stack = original_frame_stack;
633            self.0.frame_stack().get().interpreter.memory.free_child_context();
634
635            if let Ok(FrameResult::Create(create_outcome)) = result {
636                if InstructionResult::Revert == *create_outcome.instruction_result() {
637                    return (
638                        [vec![0x00], create_outcome.output().to_vec()].concat(),
639                        VecReader::new(vec![]),
640                        ArbGas(gas.spent()),
641                    );
642                }
643
644                gas.erase_cost(create_outcome.gas().remaining());
645                if let Some(address) = create_outcome.address {
646                    gas.erase_cost(create_outcome.gas().remaining() + gas_stipend);
647
648                    return (
649                        [vec![0x01], address.to_vec()].concat(),
650                        VecReader::new(vec![]),
651                        ArbGas(gas.spent()),
652                    );
653                }
654            }
655        }
656
657        error_response
658    }

660    /// Handle log emission with closure-based log handling
661    fn handle_emit_log<F>(
662        &mut self,
663        input: InputsImpl,
664        data: Vec<u8>,
665        log_handler: F,
666    ) -> (Vec<u8>, VecReader, ArbGas)
667    where
668        F: FnOnce((&mut Self, Log)),
669    {
670        let mut data = data;
671        let topic_count = buffer::take_u32(&mut data);
672        let mut topics = Vec::with_capacity(topic_count as usize);
673        for _ in 0..topic_count {
674            topics.push(buffer::take_bytes32(&mut data));
675        }
676        let log_data = buffer::take_rest(&mut data);
677
678        let log = Log::new_unchecked(input.target_address, topics, log_data);
679
680        log_handler((self, log));
681
682        (vec![], VecReader::new(vec![]), ArbGas(0))
683    }
684
685    pub(crate) fn request(
686        &mut self,
687        input: InputsImpl,
688        is_static: bool,
689        req_type: EvmApiMethod,
690        data: Vec<u8>,
691    ) -> (Vec<u8>, VecReader, ArbGas) {
692        match req_type {
693            EvmApiMethod::ContractCall | EvmApiMethod::DelegateCall | EvmApiMethod::StaticCall => {
694                self.handle_contract_call(input, is_static, req_type, data, |evm, frame_init| {
695                    evm.run_exec_loop(frame_init)
696                })
697            }
698
699            EvmApiMethod::Create1 | EvmApiMethod::Create2 => self.handle_contract_creation(
700                input,
701                is_static,
702                req_type,
703                data,
704                |evm, frame_init| evm.run_exec_loop(frame_init),
705            ),
706
707            EvmApiMethod::EmitLog => {
708                self.handle_emit_log(input, data, |(evm, log): (&mut Self, Log)| {
709                    let context = evm.ctx();
710                    context.log(log);
711                })
712            }
713
714            _ => self.request_inner(input, is_static, req_type, data),
715        }
716    }
717
718    fn request_inner(
719        &mut self,
720        input: InputsImpl,
721        is_static: bool,
722        req_type: EvmApiMethod,
723        data: Vec<u8>,
724    ) -> (Vec<u8>, VecReader, ArbGas) {
725        let context = self.ctx();
726        let mut data = data;
727
728        let spec = context.cfg().spec();
729
730        match req_type {
731            EvmApiMethod::GetBytes32 => {
732                let slot = buffer::take_u256(&mut data);
733                if let Some(result) = context.sload(input.target_address, slot) {
734                    let gas = sload_cost(spec.into(), result.is_cold);
735                    (result.to_be_bytes_vec(), VecReader::new(vec![]), ArbGas(gas))
736                } else {
737                    (vec![], VecReader::new(vec![]), ArbGas(0))
738                }
739            }
740
741            EvmApiMethod::SetTrieSlots => {
742                let gas_left = buffer::take_u64(&mut data);
743
744                if is_static {
745                    return (
746                        Status::WriteProtection.into(),
747                        VecReader::new(vec![]),
748                        ArbGas(gas_left),
749                    );
750                }
751
752                let mut total_cost = 0;
753                while !data.is_empty() {
754                    let (key, value) = (buffer::take_u256(&mut data), buffer::take_u256(&mut data));
755
756                    match context.sstore(input.target_address, key, value) {
757                        Some(result) => {
758                            total_cost +=
759                                sstore_cost(spec.clone().into(), &result.data, result.is_cold);
760
761                            if gas_left < total_cost {
762                                return (
763                                    Status::OutOfGas.into(),
764                                    VecReader::new(vec![]),
765                                    ArbGas(gas_left),
766                                );
767                            }
768                        }
769                        _ => {
770                            return (
771                                Status::Failure.into(),
772                                VecReader::new(vec![]),
773                                ArbGas(gas_left),
774                            )
775                        }
776                    }
777                }
778
779                (Status::Success.into(), VecReader::new(vec![]), ArbGas(total_cost))
780            }
781
782            EvmApiMethod::GetTransientBytes32 => {
783                let slot = buffer::take_u256(&mut data);
784                let result = context.tload(input.target_address, slot);
785                (result.to_be_bytes_vec(), VecReader::new(vec![]), ArbGas(0))
786            }
787
788            EvmApiMethod::SetTransientBytes32 => {
789                if is_static {
790                    return (Status::WriteProtection.into(), VecReader::new(vec![]), ArbGas(0));
791                }
792                let key = buffer::take_u256(&mut data);
793                let value = buffer::take_u256(&mut data);
794                context.tstore(input.target_address, key, value);
795                (Status::Success.into(), VecReader::new(vec![]), ArbGas(0))
796            }
797            EvmApiMethod::AccountBalance => {
798                let address = buffer::take_address(&mut data);
799                let balance = context.balance(address).unwrap();
800                let gas = wasm_account_touch(context, balance.is_cold, false);
801                (balance.to_be_bytes_vec(), VecReader::new(vec![]), ArbGas(gas))
802            }
803
804            EvmApiMethod::AccountCode => {
805                let address = buffer::take_address(&mut data);
806                let code = context.load_account_code(address).unwrap();
807                let gas = wasm_account_touch(context, code.is_cold, true);
808                (vec![], VecReader::new(code.to_vec()), ArbGas(gas))
809            }
810
811            EvmApiMethod::AccountCodeHash => {
812                let address = buffer::take_address(&mut data);
813                let code_hash = context.load_account_code_hash(address).unwrap();
814                let gas = wasm_account_touch(context, code_hash.is_cold, false);
815                (code_hash.to_vec(), VecReader::new(vec![]), ArbGas(gas))
816            }
817
818            EvmApiMethod::AddPages => {
819                let _count = buffer::take_u16(&mut data);
820                (Status::Success.into(), VecReader::new(vec![]), ArbGas(0))
821            }
822
823            EvmApiMethod::CaptureHostIO => {
824                //let data = buffer::take_rest(&mut data);
825                //println!("CaptureHostIO: {:?}", String::from_utf8_lossy(&data));
826                (Status::Success.into(), VecReader::new(vec![]), ArbGas(0))
827            }
828            _ => unimplemented!("EVM API method not implemented: {:?}", req_type),
829        }
830    }
831
832    /// Executes the main frame processing loop.
833    ///
834    /// This loop manages the frame stack, processing each frame until execution completes.
835    /// For each iteration:
836    /// 1. Calls the current frame
837    /// 2. Handles the returned frame input or result
838    /// 3. Creates new frames or propagates results as needed
839    #[inline]
840    fn run_exec_loop(
841        &mut self,
842        first_frame_input: FrameInit,
843    ) -> Result<FrameResult, ContextError<<<CTX as ContextTr>::Db as Database>::Error>> {
844        let res = self.frame_init(first_frame_input)?;
845
846        if let ItemOrResult::Result(frame_result) = res {
847            return Ok(frame_result);
848        }
849
850        loop {
851            let call_or_result = self.frame_run()?;
852
853            let result = match call_or_result {
854                ItemOrResult::Item(init) => {
855                    match self.frame_init(init)? {
856                        ItemOrResult::Item(_) => {
857                            continue;
858                        }
859                        // Do not pop the frame since no new frame was created
860                        ItemOrResult::Result(result) => result,
861                    }
862                }
863                ItemOrResult::Result(result) => result,
864            };
865
866            if let Some(result) = self.frame_return_result(result)? {
867                return Ok(result);
868            }
869        }
870    }
871}

904                    evm.inspect_run_exec_loop(frame_init)

927    /// Run inspection on execution loop.
928    ///
929    /// It will call:
930    /// * [`Inspector::call`],[`Inspector::create`] to inspect call, create and eofcreate.
931    /// * [`Inspector::call_end`],[`Inspector::create_end`] to inspect call, create and eofcreate
932    ///   end.
933    /// * [`Inspector::initialize_interp`] to inspect initialized interpreter.
934    fn inspect_run_exec_loop(
935        &mut self,
936        first_frame_input: FrameInit,
937    ) -> Result<FrameResult, ContextError<<<CTX as ContextTr>::Db as Database>::Error>> {
938        let res = self.inspect_frame_init(first_frame_input)?;
939
940        if let ItemOrResult::Result(frame_result) = res {
941            return Ok(frame_result);
942        }

945            let call_or_result = self.inspect_frame_run()?;
946
947            let result = match call_or_result {
948                ItemOrResult::Item(init) => {
949                    match self.inspect_frame_init(init)? {
950                        ItemOrResult::Item(_) => {
951                            continue;
952                        }
953                        // Do not pop the frame since no new frame was created
954                        ItemOrResult::Result(result) => result,
955                    }
956                }
957                ItemOrResult::Result(result) => result,
958            };
959
960            if let Some(result) = self.frame_return_result(result)? {
961                return Ok(result);
962            }
963        }
964    }
