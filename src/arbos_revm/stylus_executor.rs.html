65pub fn build_evm_data<CTX>(context: &mut CTX, input: InputsImpl) -> EvmData
66where
67    CTX: ArbitrumContextTr,
68{
69    let config_env = context.cfg();
70    let arbos_env = context.chain();
71
72    let block_env = context.block();
73    let tx_env = context.tx();
74
75    let base_fee = block_env.basefee();
76
77    let evm_data: EvmData = EvmData {
78        arbos_version: arbos_env.arbos_version() as u64,
79        block_basefee: Bytes32::from(U256::from(base_fee).to_be_bytes()),
80        chainid: config_env.chain_id(),
81        block_coinbase: Bytes20::try_from(block_env.beneficiary().as_slice()).unwrap(),
82        block_gas_limit: U64::wrapping_from(block_env.gas_limit()).to::<u64>(),
83        block_number: U64::wrapping_from(block_env.number()).to::<u64>(),
84        block_timestamp: U64::wrapping_from(block_env.timestamp()).to::<u64>(),
85        contract_address: Bytes20::try_from(input.target_address.as_slice()).unwrap(),
86        module_hash: Bytes32::try_from(keccak256(input.target_address.as_slice()).as_slice())
87            .unwrap(),
88        msg_sender: Bytes20::try_from(input.caller_address.as_slice()).unwrap(),
89        msg_value: Bytes32::try_from(input.call_value.to_be_bytes_vec()).unwrap(),
90        tx_gas_price: Bytes32::from(
91            U256::from(tx_env.effective_gas_price(base_fee as u128)).to_be_bytes(),
92        ),
93        tx_origin: Bytes20::try_from(tx_env.caller().as_slice()).unwrap(),
94        reentrant: 0,
95        return_data_len: 0,
96        cached: true,
97        tracing: true,
98    };
99
100    evm_data
101}

114pub fn stylus_call_cost(new: u16, open: u16, ever: u16) -> u64 {
115    let new_open = open.saturating_add(new);
116    let new_ever = max(ever, new_open);
117
118    if new_ever < INITIAL_FREE_PAGES as u16 {
119        return 0;
120    }
121
122    let adding = new_open.saturating_sub(open).saturating_sub(INITIAL_FREE_PAGES as u16);
123    let linear = (adding as u64).saturating_mul(INITIAL_PAGE_GAS);
124    let exp = |x: u16| -> u64 {
125        if x < MEMORY_EXPONENTS.len() as u16 {
126            return MEMORY_EXPONENTS[x as usize] as u64;
127        }
128
129        u64::MAX
130    };
131
132    let expand = exp(new_ever) - exp(ever);
133
134    linear.saturating_add(expand)
135}

137pub fn init_gas(params: StylusData) -> u64 {
138    let base = MIN_INIT_GAS_UNITS * INITIAL_MIN_INIT_GAS;
139    let dyno = (params.init_cost as u64)
140        .saturating_mul(INITIAL_INIT_COST_SCALAR as u64 * COST_SCALAR_PERCENT);
141    base.saturating_add(dyno.div_ceil(100))
142}
143
144pub fn cached_gas(params: StylusData) -> u64 {
145    let base = INITIAL_MIN_CACHED_GAS * MIN_CACHED_GAS_UNITS;
146    let dyno = (params.cached_init_cost as u64)
147        .saturating_mul(INITIAL_CACHED_COST_SCALAR as u64 * COST_SCALAR_PERCENT);
148    base.saturating_add(dyno.div_ceil(100))
149}

157    /// Common method to build API requestor for both inspected and non-inspected modes
158    fn build_api_requestor(
159        &mut self,
160        input: InputsImpl,
161        is_static: bool,
162        request_handler: impl Fn(
163            &mut Self,
164            InputsImpl,
165            bool,
166            EvmApiMethod,
167            Vec<u8>,
168        ) -> (Vec<u8>, VecReader, ArbGas),
169    ) -> EvmApiRequestor<VecReader, StylusHandler> {
170        let evm = Arc::new(Mutex::new(self));
171
172        let callback = {
173            let evm = evm.clone();
174
175            move |req_type: arbutil::evm::api::EvmApiMethod,
176                  req_data: Vec<u8>|
177                  -> (Vec<u8>, VecReader, arbutil::evm::api::Gas) {
178                let mut evm = evm.lock().unwrap();
179                request_handler(&mut evm, input.clone(), is_static, req_type, req_data)
180            }
181        };
182
183        let callback: EvmApiHandler<'_> = Arc::new(Box::new(callback));
184        let unsafe_callback: &'static EvmApiHandler<'_> = unsafe { mem::transmute(&callback) };
185        EvmApiRequestor::new(StylusHandler::new(unsafe_callback.clone()))
186    }

188    /// Extract common Stylus execution context from frame input
189    fn extract_stylus_context(&mut self) -> Option<(StylusExecutionContext, B256)> {
190        let frame_input = {
191            let frame = self.frame_stack().get();
192            match frame.input {
193                FrameInput::Call(ref input) => input.clone(),
194                _ => return None,
195            }
196        };
197
198        let bytecode_address = frame_input.bytecode_address;
199
200        let code_hash = {
201            let ctx = self.ctx();
202            if let Ok(code_hash) = ctx.journal_mut().code_hash(bytecode_address) {
203                code_hash.data
204            } else {
205                return None;
206            }
207        };
208
209        let calldata = match &frame_input.input {
210            CallInput::Bytes(calldata) => calldata.clone(),
211            CallInput::SharedBuffer(range) => {
212                if let Some(slice) = self.ctx().local().shared_memory_buffer_slice(range.clone()) {
213                    Bytes::from(slice.to_vec())
214                } else {
215                    Bytes::new()
216                }
217            }
218        };
219
220        let context = StylusExecutionContext {
221            target_address: frame_input.target_address,
222            bytecode_address,
223            caller_address: frame_input.caller,
224            call_value: frame_input.value.get(),
225            is_static: frame_input.is_static,
226            gas_limit: frame_input.gas_limit,
227            calldata,
228        };
229
230        Some((context, code_hash))
231    }

233    /// Compile Stylus bytecode
234    fn compile_stylus_bytecode(
235        bytecode: &Bytes,
236        code_hash: B256,
237        arbos_version: u16,
238        stylus_version: u16,
239        _gas_limit: u64,
240        compile_config: &CompileConfig,
241    ) -> Result<(Vec<u8>, Module, StylusData), ()> {
242        if let Some(bytecode) = bytecode.strip_prefix(STYLUS_DISCRIMINANT) {
243            let (dictionary, compressed_bytecode) =
244                if let Some((dictionary, compressed_bytecode)) = bytecode.split_at_checked(1) {
245                    (dictionary, compressed_bytecode)
246                } else {
247                    return Err(());
248                };
249
250            let dictionary = match dictionary[0] {
251                0x00 => Dictionary::Empty,
252                0x01 => Dictionary::StylusProgram,
253                _ => unreachable!(),
254            };
255
256            let bytecode = brotli::decompress(compressed_bytecode, dictionary)
257                .or_else(|err| {
258                    if dictionary == Dictionary::Empty {
259                        Ok(compressed_bytecode.to_vec())
260                    } else {
261                        Err(err)
262                    }
263                })
264                .unwrap();
265
266            let mut activation_gas = 10_000_000u64;
267            let (module, stylus_data) = native::activate(
268                bytecode.as_slice(),
269                &Bytes32::from(code_hash.0),
270                stylus_version,
271                arbos_version as u64,
272                128,
273                false,
274                &mut activation_gas,
275            )
276            .unwrap();
277
278            let bytecode = native::compile(
279                bytecode.as_slice(),
280                compile_config.version,
281                false,
282                wasmer_types::compilation::target::Target::default(),
283                true,
284            )
285            .unwrap();
286
287            return Ok((bytecode, module, stylus_data));
288        }
289
290        Err(())
291    }

293    /// Core Stylus execution logic shared between inspected and non-inspected modes
294    pub(crate) fn execute_stylus_program(
295        &mut self,
296        stylus_ctx: StylusExecutionContext,
297        code_hash: B256,
298        api_request_handler: impl Fn(
299            &mut Self,
300            InputsImpl,
301            bool,
302            EvmApiMethod,
303            Vec<u8>,
304        ) -> (Vec<u8>, VecReader, ArbGas),
305    ) -> Option<InterpreterAction> {
306        let context = self.ctx();
307
308        let compile_config =
309            CompileConfig::version(context.chain().stylus_version(), context.chain().debug_mode());
310
311        let stylus_config = StylusConfig::new(
312            context.chain().stylus_version(),
313            context.chain().max_depth(),
314            context.chain().ink_price(),
315        );
316
317        let (serialized, _module, stylus_data) = {
318            // Use read lock to get cached program if available
319            // if not available drop the read lock and acquire write lock to compile and insert
320            let maybe_cached = {
321                let mut cache = PROGRAM_CACHE.lock().unwrap();
322                if let Some((serialized, module, stylus_data)) = cache.get(&code_hash).cloned() {
323                    Some((serialized, module, stylus_data))
324                } else {
325                    None
326                }
327            };
328
329            if let Some((serialized, module, stylus_data)) = maybe_cached {
330                (serialized, module, stylus_data)
331            } else {
332                let bytecode = context.journal_mut().code(stylus_ctx.bytecode_address).ok()?.data;
333
334                if !bytecode.starts_with(STYLUS_DISCRIMINANT) {
335                    return None;
336                }
337
338                if let Ok((serialized, module, stylus_data)) = Self::compile_stylus_bytecode(
339                    &bytecode,
340                    code_hash,
341                    context.chain().arbos_version(),
342                    context.chain().stylus_version(),
343                    stylus_ctx.gas_limit,
344                    &compile_config,
345                ) {
346                    let mut cache = PROGRAM_CACHE.lock().unwrap();
347                    cache.get_or_insert(code_hash, || {
348                        (serialized.clone(), module.clone(), stylus_data)
349                    });
350
351                    (serialized, module, stylus_data)
352                } else {
353                    return None;
354                }
355            }
356        };
357
358        let inputs = InputsImpl {
359            target_address: stylus_ctx.target_address,
360            caller_address: stylus_ctx.caller_address,
361            input: CallInput::Bytes(Bytes::from(stylus_ctx.calldata.to_vec())),
362            call_value: stylus_ctx.call_value,
363            bytecode_address: Some(stylus_ctx.target_address),
364        };
365
366        let call_cost = stylus_call_cost(stylus_data.footprint, 0, INITIAL_FREE_PAGES as u16)
367            + cached_gas(stylus_data);
368
369        let mut gas = Gas::new(stylus_ctx.gas_limit);
370        if !gas.record_cost(call_cost) {
371            return Some(InterpreterAction::Return(InterpreterResult {
372                result: InstructionResult::OutOfGas,
373                output: Default::default(),
374                gas: Default::default(),
375            }));
376        }
377
378        let evm_data = build_evm_data(self.ctx(), inputs.clone());
379        let evm_api =
380            self.build_api_requestor(inputs.clone(), stylus_ctx.is_static, api_request_handler);
381
382        let mut instance = unsafe {
383            NativeInstance::deserialize(serialized.as_slice(), compile_config, evm_api, evm_data)
384                .unwrap()
385        };
386
387        let ink_limit = stylus_config.pricing.gas_to_ink(arbutil::evm::api::Gas(gas.remaining()));
388        gas.spend_all();
389
390        let bytecode = match inputs.input() {
391            CallInput::Bytes(bytes) => bytes,
392            CallInput::SharedBuffer(_) => todo!(),
393        };
394
395        let outcome = match instance.run_main(bytecode, stylus_config, ink_limit) {
396            Err(e) | Ok(UserOutcome::Failure(e)) => UserOutcome::Failure(e.wrap_err("call failed")),
397            Ok(outcome) => outcome,
398        };
399
400        let mut gas_left = stylus_config.pricing.ink_to_gas(instance.ink_left().into()).0;
401
402        let (kind, data) = outcome.into_data();
403
404        let result = match kind {
405            UserOutcomeKind::Success => revm::interpreter::InstructionResult::Return,
406            UserOutcomeKind::Revert => revm::interpreter::InstructionResult::Revert,
407            UserOutcomeKind::Failure => revm::interpreter::InstructionResult::Revert,
408            UserOutcomeKind::OutOfInk => revm::interpreter::InstructionResult::OutOfGas,
409            UserOutcomeKind::OutOfStack => {
410                gas_left = 0;
411                revm::interpreter::InstructionResult::StackOverflow
412            }
413        };
414
415        gas.erase_cost(gas_left);
416
417        Some(InterpreterAction::Return(InterpreterResult { result, output: data.into(), gas }))
418    }

420    pub fn frame_run_stylus(&mut self) -> Option<InterpreterAction> {
421        let (stylus_ctx, code_hash) = self.extract_stylus_context()?;
422        self.execute_stylus_program(
423            stylus_ctx,
424            code_hash,
425            |evm, inputs, is_static, req_type, data| evm.request(inputs, is_static, req_type, data),
426        )
427    }

450    pub(crate) fn inspect_request(
451        &mut self,
452        input: InputsImpl,
453        is_static: bool,
454        req_type: EvmApiMethod,
455        data: Vec<u8>,
456    ) -> (Vec<u8>, VecReader, ArbGas) {
457        match req_type {
458            EvmApiMethod::ContractCall | EvmApiMethod::DelegateCall | EvmApiMethod::StaticCall => {
459                self.handle_contract_call(input, is_static, req_type, data, |evm, frame_init| {
460                    evm.inspect_run_exec_loop(frame_init)
461                })
462            }
463
464            EvmApiMethod::Create1 | EvmApiMethod::Create2 => self.handle_contract_creation(
465                input,
466                is_static,
467                req_type,
468                data,
469                |evm, frame_init| evm.inspect_run_exec_loop(frame_init),
470            ),
471
472            EvmApiMethod::EmitLog => {
473                self.handle_emit_log(input, data, |(evm, log): (&mut Self, Log)| {
474                    let (context, inspector, frame) = evm.ctx_inspector_frame();
475                    context.log(log.clone());
476                    inspector.log(&mut frame.interpreter, context, log);
477                })
478            }
479            _ => self.request_inner(input, is_static, req_type, data),
480        }
481    }
