35impl<CTX, INSP, P, I> InspectorEvmTr for ArbitrumEvm<CTX, INSP, P, I>
36where
37    CTX: ArbitrumContextTr<Journal: JournalExt> + ContextSetters,
38    I: InstructionProvider<Context = CTX, InterpreterTypes = EthInterpreter>,
39    P: PrecompileProvider<CTX, Output = InterpreterResult>,
40    INSP: Inspector<CTX, I::InterpreterTypes>,
41{
42    type Inspector = INSP;
43
44    fn inspector(&mut self) -> &mut Self::Inspector {
45        &mut self.0.inspector
46    }
47
48    fn ctx_inspector(&mut self) -> (&mut Self::Context, &mut Self::Inspector) {
49        (&mut self.0.ctx, &mut self.0.inspector)
50    }
51
52    fn ctx_inspector_frame(
53        &mut self,
54    ) -> (&mut Self::Context, &mut Self::Inspector, &mut Self::Frame) {
55        (&mut self.0.ctx, &mut self.0.inspector, self.0.frame_stack.get())
56    }
57
58    fn ctx_inspector_frame_instructions(
59        &mut self,
60    ) -> (&mut Self::Context, &mut Self::Inspector, &mut Self::Frame, &mut Self::Instructions) {
61        (&mut self.0.ctx, &mut self.0.inspector, self.0.frame_stack.get(), &mut self.0.instruction)
62    }
63
64    /// Run the frame from the top of the stack. Returns the frame init or result.
65    ///
66    /// If frame has returned result it would mark it as finished.
67    #[inline]
68    fn inspect_frame_run(
69        &mut self,
70    ) -> Result<FrameInitOrResult<Self::Frame>, ContextDbError<Self::Context>> {
71        if let Some(next_action) = self.inspect_frame_run_stylus() {
72            let frame = self.0.frame_stack.get();
73            let context = &mut self.0.ctx;
74            let mut result = frame.process_next_action(context, next_action);
75
76            if let Ok(ItemOrResult::Result(frame_result)) = &mut result {
77                let (ctx, inspector, frame) = self.ctx_inspector_frame();
78                frame_end(ctx, inspector, frame.frame_input(), frame_result);
79                frame.set_finished(true);
80            };
81            return result;
82        }
83
84        self.0.inspect_frame_run()
85    }
