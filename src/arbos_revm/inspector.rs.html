37impl<CTX, INSP, P, I> InspectorEvmTr for ArbitrumEvm<CTX, INSP, P, I>
38where
39    CTX: ArbitrumContextTr<Journal: JournalExt> + ContextSetters,
40    I: InstructionProvider<Context = CTX, InterpreterTypes = EthInterpreter>,
41    P: PrecompileProvider<CTX, Output = InterpreterResult>,
42    INSP: Inspector<CTX, I::InterpreterTypes>,
43{
44    type Inspector = INSP;
45
46    fn inspector(&mut self) -> &mut Self::Inspector {
47        &mut self.0.inspector
48    }
49
50    fn ctx_inspector(&mut self) -> (&mut Self::Context, &mut Self::Inspector) {
51        (&mut self.0.ctx, &mut self.0.inspector)
52    }
53
54    fn ctx_inspector_frame(
55        &mut self,
56    ) -> (&mut Self::Context, &mut Self::Inspector, &mut Self::Frame) {
57        (&mut self.0.ctx, &mut self.0.inspector, self.0.frame_stack.get())
58    }
59
60    fn ctx_inspector_frame_instructions(
61        &mut self,
62    ) -> (&mut Self::Context, &mut Self::Inspector, &mut Self::Frame, &mut Self::Instructions) {
63        (&mut self.0.ctx, &mut self.0.inspector, self.0.frame_stack.get(), &mut self.0.instruction)
64    }
65
66    /// Run the frame from the top of the stack. Returns the frame init or result.
67    ///
68    /// If frame has returned result it would mark it as finished.
69    #[inline]
70    fn inspect_frame_run(
71        &mut self,
72    ) -> Result<FrameInitOrResult<Self::Frame>, ContextDbError<Self::Context>> {
73        if let Some(next_action) = self.inspect_frame_run_stylus() {
74            let frame = self.0.frame_stack.get();
75            let context = &mut self.0.ctx;
76            let mut result = frame.process_next_action(context, next_action);
77
78            if let Ok(ItemOrResult::Result(frame_result)) = &mut result {
79                let (ctx, inspector, frame) = self.ctx_inspector_frame();
80                frame_end(ctx, inspector, frame.frame_input(), frame_result);
81                frame.set_finished(true);
82            };
83            return result;
84        }
85
86        self.0.inspect_frame_run()
87    }
88}

90impl<CTX, INSP, P, I> ArbitrumEvm<CTX, INSP, P, I>
91where
92    CTX: ArbitrumContextTr,
93    CTX::Journal: JournalExt,
94    I: InstructionProvider<Context = CTX, InterpreterTypes = EthInterpreter>,
95    P: PrecompileProvider<CTX, Output = InterpreterResult>,
96    CTX: ContextSetters,
97    INSP: Inspector<CTX>,
98{
99    /// Run inspection on execution loop.
100    ///
101    /// It will call:
102    /// * [`Inspector::call`],[`Inspector::create`] to inspect call, create and eofcreate.
103    /// * [`Inspector::call_end`],[`Inspector::create_end`] to inspect call, create and eofcreate
104    ///   end.
105    /// * [`Inspector::initialize_interp`] to inspect initialized interpreter.
106    pub(crate) fn inspect_run_exec_loop(
107        &mut self,
108        first_frame_input: FrameInit,
109    ) -> Result<FrameResult, ContextError<<<CTX as ContextTr>::Db as Database>::Error>> {
110        let res = self.inspect_frame_init(first_frame_input)?;
111
112        if let ItemOrResult::Result(frame_result) = res {
113            return Ok(frame_result);
114        }
115
116        loop {
117            let call_or_result = self.inspect_frame_run()?;
118
119            let result = match call_or_result {
120                ItemOrResult::Item(init) => {
121                    match self.inspect_frame_init(init)? {
122                        ItemOrResult::Item(_) => {
123                            continue;
124                        }
125                        // Do not pop the frame since no new frame was created
126                        ItemOrResult::Result(result) => result,
127                    }
128                }
129                ItemOrResult::Result(result) => result,
130            };
131
132            if let Some(result) = self.frame_return_result(result)? {
133                return Ok(result);
134            }
135        }
136    }
