11pub struct ArbitrumHandler<EVM, ERROR, FRAME> {
12    /// Mainnet handler allows us to use functions from the mainnet handler inside optimism
13    /// handler. So we dont duplicate the logic
14    pub mainnet: MainnetHandler<EVM, ERROR, FRAME>,
15    /// Phantom data to avoid type inference issues.
16    pub _phantom: core::marker::PhantomData<(EVM, ERROR, FRAME)>,
17}
18
19impl<EVM, ERROR, FRAME> ArbitrumHandler<EVM, ERROR, FRAME> {

21        Self { mainnet: MainnetHandler::default(), _phantom: core::marker::PhantomData }
22    }
23}
